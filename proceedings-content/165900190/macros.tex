% Stacked hypotheses for ebproof

% Taken from https://tex.stackexchange.com/a/691409
% Usage:
%   #1 : Number of \hypo in line 1 
%   #2 : line 1, it probably will do weird things if its not just a series of \hypo 
%   #3 : Number of \hypo in line 2
%   #4 : line 2, same note as above
\newcommand{\stackedhypos}[4]{%
    \hypo{
        \begin{prooftree}
            #2 % line 1
            \infer[no rule]#1{ 
                \begin{prooftree}[rule margin=0ex]
                    #4 % line 2
                    \infer[no rule]#3{}
                \end{prooftree}
            }
        \end{prooftree}
    }
}

% redefine the \mess, otherwise can't use mathmode in labels of messages
% https://tex.stackexchange.com/a/164725
\renewcommand{\mess}[4][0]{
  \stepcounter{seqlevel}
  \path
  (#2)+(0,-\theseqlevel*\unitfactor-0.7*\unitfactor) node (mess from) {};
  \addtocounter{seqlevel}{#1}
  \path
  (#4)+(0,-\theseqlevel*\unitfactor-0.7*\unitfactor) node (mess to) {};
  \draw[->,>=angle 60] (mess from) -- (mess to) node[midway, above]
  {#3};
}
